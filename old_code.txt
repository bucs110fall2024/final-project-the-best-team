def on_screen(x,y):
    if((x<600 and x>0) and(y<600 and y>0)):
        return True
    return False
def detect_collision(collider,collided):
    if((collided.x+50>collider.x and collider.x>collided.x)and(collided.y+50>collider.y and collider.y>collided.y)):
        return True
    return False
